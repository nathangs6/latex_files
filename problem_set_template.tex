% A generic template for writing problem set solutions / notes in.
\documentclass{article}
% Formatting settings
\usepackage[left=3cm,right=3cm,top=2cm,bottom=2cm]{geometry}
\setlength{\parindent}{0mm}  % Remove paragraph indenting


\usepackage{amsmath}  % Provides a lot of math options
\usepackage{amsthm}  % Used for theorem styles and proof environment
\usepackage{amssymb}  % Provides a lot more math symbols

%%% Setup Math Related Environments %%%
% Theorem related styles
\theoremstyle{plain}
\newtheorem{thm}{Theorem}
\newtheorem{lem}[thm]{Lemma}
\newtheorem{prop}[thm]{Proposition}
\newtheorem*{cor}{Corollary}
\newtheorem*{claim}{Claim}

% Definition related styles
\theoremstyle{definition}
\newtheorem{defn}{Definition}
\newtheorem{exmp}{Example}

% Remark related styles
\theoremstyle{remark}
\newtheorem*{rem}{Remark}
\newtheorem*{exercise}{Exercise}

% Define proof and solution environments
\newenvironment{solution}{\textit{Solution:}}{\hfill$\square$ \\}
\newenvironment{inline_proof}{\textit{Proof:}}{\hfill$\blacksquare$ \\}
\newenvironment{claim_proof}{\textit{Proof of Claim:}}{\hfill$\framebox(5.166648,5.166648){\rule{1.2ex}{0.6ex}}$ \\}

% Change qed symbol to the superior blacksquare
\renewcommand{\qed}{\hfill$\blacksquare$}


%%% Setup math related symbols and operators %%%
% Define some symbol shortcuts
\newcommand{\C}{\mathbb{C}}
\newcommand{\R}{\mathbb{R}}
\newcommand{\Q}{\mathbb{Q}}
\newcommand{\Z}{\mathbb{Z}}
\newcommand{\N}{\mathbb{N}}

% Declare math operators
% \DeclareMathOperator{\command}{output}


%%% Setup non-math related environments %%%
% Setup title
\newcommand\doctitle[2]{  % Arguments go {course code}{assignment number}
	\begin{center}
		\begin{LARGE}
			#1 Assignment \##2
		\end{LARGE}
	\end{center}
}
% Setup question environment
\usepackage{tcolorbox}
\newenvironment{question}[1]{\begin{tcolorbox}\textbf{Question #1: }}{\end{tcolorbox}}

% Other useful packages
\usepackage{enumitem}  % Allows for changing the labels of enumerate lists
